 38   # 
 39   #   WARN loggername(file.rb at 12): Hot potato!
 40   #

 42   # +inspect+ everything else.

 52       @depth = (hash[:depth] or hash['depth'] or 7).to_i
 53     end

 55     def format(event)
 56       buff = sprintf(@@basicformat, MaxLevelLength, LNAMES[event.level],
 57              event.name)

 59       buff += format_object(event.data) + "\n"
 60       buff
 61     end

 63     # Formats data according to its class:
 64     #
 65     # String::     Prints it out as normal.
 66     # Exception::  Produces output similar to command-line exceptions.
 67     # Object::     Prints the type of object, then the output of
 68     #              +inspect+. An example -- Array: [1, 2, 3]
 69 
 70     def format_object(obj)
 71       if obj.kind_of? Exception
 72         return "Caught #{obj.class}: #{obj.message}\n\t" +\
 73                obj.backtrace[0...@depth].join("\n\t")
 74       elsif obj.kind_of? String
 75         return obj
 76       else # inspect the object
 77         return "#{obj.class}: #{obj.inspect}"
 78       end
 79     end
 80   end

 82   # Formats objects the same way irb does:
 83   #
 84   #   loggername:foo.rb in 12> 
 85   #   [1, 3, 4]
 86   #   loggername:foo.rb in 13> 
 87   #   {1=>"1"}

 89   # Strings don't get inspected. just printed. The trace is optional.

 91   class ObjectFormatter < Formatter
 92     def format(event)
 93       buff = event.logger.name
 94       buff += (event.tracer.nil? ? "" : ":#{event.tracer[0]}") + ">\n"
 95       buff += (event.data.kind_of?(String) ? event.data : event.data.inspect)
 96       buff += "\n"
 97     end
 98   end
